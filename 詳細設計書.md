---

🟩 Pseudo K-means（軽量版）

ステップ1：

summary / fullOpinion をベクトル化（超簡易版）

• 文字数
• 句読点の数
• 名詞の数（簡易カウント）
• ひらがな比率
• カタカナ比率
• 英数字比率


→ これだけで クラスタリングっぽい特徴量 が作れる。

ステップ2：

初期クラスタ中心をランダムに3つ作る

ステップ3：

各点を最も近い中心に割り当てる

ステップ4：

中心を更新する（2〜3回だけ繰り返す）

ステップ5：

クラスタごとに座標を生成（あなたの scatter.ts と同じ）

• クラスタ A → 左上
• クラスタ B → 右上
• クラスタ C → 下側
• • ランダムノイズ

• • クラスタごとの重心調整

🟩 server.py（クラスタリング付き・完全版）

import csv
import sqlite3
import random
import math
from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware
import uvicorn
import os

DB_PATH = "data.db"
CSV_PATH = "data30.csv"

app = FastAPI()

app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_methods=["*"],
    allow_headers=["*"],
)

# ---------------------------------------------------
# 文字列 → 簡易ベクトル化（特徴量抽出）
# ---------------------------------------------------
def text_to_vector(text: str):
    length = len(text)
    num_punct = sum(1 for c in text if c in "。、.,!?！？")
    num_hira = sum(1 for c in text if "ぁ" <= c <= "ん")
    num_kata = sum(1 for c in text if "ァ" <= c <= "ン")
    num_digit = sum(1 for c in text if c.isdigit())

    return [
        length / 500,          # 長さ
        num_punct / 50,        # 句読点
        num_hira / length,     # ひらがな比率
        num_kata / length,     # カタカナ比率
        num_digit / length     # 数字比率
    ]

# ---------------------------------------------------
# 簡易 k-means（3クラスタ固定）
# ---------------------------------------------------
def pseudo_kmeans(vectors, k=3, iterations=3):
    # 初期中心をランダムに選ぶ
    centers = random.sample(vectors, k)

    for _ in range(iterations):
        clusters = [[] for _ in range(k)]

        # 割り当て

            dists = [sum((v[i] - c[i])**2 for i in range(len(v))) for c in centers]
            idx = dists.index(min(dists))
            clusters[idx].append(v)

        # 中心更新
        new_centers = []
        for group in clusters:
            if not group:
                new_centers.append(random.choice(vectors))
            else:
                dim = len(group[0])
                new_centers.append([
                    sum(v[i] for v in group) / len(group)
                    for i in range(dim)
                ])
        centers = new_centers

    # 最終割り当て
    labels = []
    for v in vectors:
        dists = [sum((v[i] - c[i])**2 for i in range(len(v))) for c in centers]
        labels.append(dists.index(min(dists)))

    return labels

# ---------------------------------------------------
# クラスタごとに座標を生成（あなたの scatter.ts と同じ思想）
# ---------------------------------------------------
def generate_xy(cluster_id):
    if cluster_id == 0:   # A
        base_x, base_y = -0.5, 0.5
    elif cluster_id == 1: # B
        base_x, base_y = 0.5, 0.5
    else:                 # C
        base_x, base_y = 0.0, -0.5

    return (
        base_x + random.uniform(-0.2, 0.2),
        base_y + random.uniform(-0.2, 0.2)
    )

# ---------------------------------------------------
# DB 初期化
# ---------------------------------------------------
def init_db():
    if os.path.exists(DB_PATH):
        print("[INFO] DB already exists. Skipping init.")
        return

    print("[INFO] Initializing DB...")
    conn = sqlite3.connect(DB_PATH)
    cur = conn.cursor()

    cur.execute("""
        CREATE TABLE opinions (
            id TEXT PRIMARY KEY,
            cluster_id TEXT,
            x REAL,
            y REAL,
            summary TEXT,
            fullOpinion TEXT
        )
    """)

    # CSV 読み込み
    rows = []
    with open(CSV_PATH, encoding="utf-8") as f:
        reader = csv.reader(f)
        next(reader)
        for row in reader:
            id_, cluster_id, x, y, summary, *rest = row
            fullOpinion = ",".join(rest)
            rows.append((id_, summary, fullOpinion))

    # ベクトル化
    vectors = [text_to_vector(summary + fullOpinion) for (_, summary, fullOpinion) in rows]

    # クラスタリング
    labels = pseudo_kmeans(vectors, k=3)

    # DB 保存
    for (id_, summary, fullOpinion), label in zip(rows, labels):
        x, y = generate_xy(label)
        cluster_name = ["A", "B", "C"][label]

        cur.execute("""
            INSERT INTO opinions (id, cluster_id, x, y, summary, fullOpinion)
            VALUES (?, ?, ?, ?, ?, ?)
        """, (id_, cluster_name, x, y, summary, fullOpinion))

    conn.commit()
    conn.close()
    print("[INFO] DB initialized with pseudo k-means clustering.")

# ---------------------------------------------------
# API
# ---------------------------------------------------
@app.get("/scatter")
def get_scatter():
    conn = sqlite3.connect(DB_PATH)
    cur = conn.cursor()

    cur.execute("SELECT id, cluster_id, x, y, summary, fullOpinion FROM opinions")
    rows = cur.fetchall()
    conn.close()

    data = []
    for r in rows:
        data.append({
            "id": r[0],
            "cluster_id": r[1],
            "x": r[2],
            "y": r[3],
            "summary": r[4],
            "fullOpinion": r[5]
        })

    return {"count": len(data), "data": data}

# ---------------------------------------------------
# 起動
# ---------------------------------------------------
if __name__ == "__main__":
    init_db()
    uvicorn.run(app, host="0.0.0.0", port=8000)
